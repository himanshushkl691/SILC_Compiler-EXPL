type
User
{
	int roll;
	str name;
}
List
{
	int a;
	User b;
	List next;
}
endtype

decl
	List insert(List l,int a,int roll,str name);
	int printList(List l);
enddecl

List insert(List l,int a,int roll,str name)
{
	decl
		List curr,new,prev;
		User user;
	enddecl
	begin
		user = alloc();
		user.roll = roll;
		user.name = name;
		new = alloc();
		new.a = a;
		new.b = user;
		new.next = null;
		curr = l;
		prev = null;
		while(curr != null) do
			prev = curr;
			curr = curr.next;
		endwhile;
		if(prev == null) then
			l = new;
		else
			prev.next = new;
		endif;
		return l;
	end
}

int printList(List l)
{
	decl
		List curr;
	enddecl
	begin
		curr = l;
		while(curr != null) do
			write("*");
			write(curr.a);
			write(curr.b.roll);
			write(curr.b.name);
			curr = curr.next;
		endwhile;
		return 0;
	end
}

int main()
{
	decl
		int ret,i,a;
		str b;
		List lst;
	enddecl
	begin
		initialize();
		lst = null;
		read(i);
		ret = 0;
		while(ret != i) do
			read(a);
			read(b);
			lst = insert(lst,ret + 1,a,b);
			ret = ret + 1;
		endwhile;
		write("First");
		ret = printList(lst);
		ret = free(lst);
		read(i);
		ret = 0;
		while(ret != i) do
			read(a);
			read(b);
			lst = insert(lst,ret + 1,a,b);
			ret = ret + 1;
		endwhile;
		write("Second");
		ret = printList(lst);
		return 0;
	end
}
